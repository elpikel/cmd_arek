defmodule AccessTokenCache do
  use GenServer

  alias AccessToken

  @doc """
    Starts the cache.
  """
  def start_link(opts) do
    GenServer.start_link(__MODULE__, :ok, opts)
  end

  @doc """
    Gets access token
  """
  def get(config) do
    GenServer.call(__MODULE__, {:get, config})
  end

  @impl true
  def init(:ok) do
    {:ok, nil}
  end

  @impl true
  def handle_call({:get, config}, _from, nil) do
    get_access_token(config)
  end

  def handle_call({:get, config}, _from, %AccessToken{} = access_token) do
    if AccessToken.valid?(access_token) do
      {:reply, {:ok, access_token}, access_token}
    else
      get_access_token(config)
    end
  end

  defp get_access_token(config) do
    case config.client_module.access_token(config) do
      {:ok, access_token} ->
        {:reply, {:ok, access_token}, access_token}

      error ->
        {:reply, error, nil}
    end
  end
end
                        
                        
                        
defmodule AccessToken do
  @enforce_keys [:token, :expires_at]
  defstruct token: nil, expires_at: nil

  def new(response) do
    response = Jason.decode!(response)
    token = response["access_token"]
    expires_at = DateTime.add(DateTime.utc_now(), response["expires_in"])

    %__MODULE__{token: token, expires_at: expires_at}
  end

  def valid?(%__MODULE__{} = access_token) do
    DateTime.compare(DateTime.add(access_token.expires_at, -60), DateTime.utc_now()) == :gt
  end
end

                        
                        

def call_to_external_api(config) do
  case AccessTokenCache.get(config) do
    {:ok, access_token} ->
      header = [
          {"Content-Type", "application/json"},
          {"Authorization", "Bearer #{access_token.token}"}
        ]
    case HTTPoison.get(url, header, recv_timeout: timeout) do
      {:ok, %HTTPoison.Response{body: body, status_code: 200}} ->
        Jason.decode(body)

      {:ok, %HTTPoison.Response{body: body, status_code: code}} ->
        {:error, body}

      {:error, %HTTPoison.Error{reason: reason}} ->
        Logger.error("GET url=#{url} failed, reason=#{reason}")
        {:error, reason}
    end

    error ->
      error
  end
end
